<script>
$(function () {
	$("a").click(function () {
		$("img").attr("src", $(this).attr("href")).attr("alt", $(this).text());
		return false;	//これをつけないと画面上での変化を確認できない（画像のページに移動してしまうため）
		//$(this)は「イベントが発生した要素」のこと。今回は「クリックしたaタグ」のこと。
		//imgタグにsrc="aタグのhrefの中身"とalt="aタグのtext部分"を追加。
	});
});
</script>
