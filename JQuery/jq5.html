色の値を入力：

http://www.htmq.com/color/colorname.shtmlを参考に入力してください。

<div id="box"></div>
<input type="text" id="color_num">
<input type="button" value="Click" onclick="colorChange()">

<script>
	function colorChange() {
		$("<div></div>").css("background-color", color_num.value).html(color_num.value).appendTo('#box');
		//.css("背景色",入力フォームのid.値)で背景色を変更
		//.html(入力フォームのid.値)で入力名称を追加。
		//.appendTo(id名)で[id名]の部分にこれまでの内容を追加。
		/*appendTo(content)
		要素の中身を他の要素に追加する。
		例えば $(A).append(B) とした場合にAにBが追加されるのに対して、$(A).appendTo(B) ではBにAが追加される。
		*/
	}
</script>
