<script>
	function imgExtend() {
		let wd = $("a>img").width();	//これによって手に入る長さはpx単位であることが分かります。
		let hg = $("a>img").height();
		console.log("横幅は" + wd + "pxです。\n" + "縦幅は" + hg + "pxです。");
		$("a>img").width("+=100").height(hg + 100);	//変数で現在の幅を取得せずとも[+=x]で現在値よりもx.px大きくすることが出来ます。
	}
	//現在の値に+100をしています。100px大きくなっていることが判明しました。

	/*
	.width(value)
	幅をvalueに設定します。valueを150などのように単位を省略した場合は、
	pxが自動で設定されます。'50%'などのように単位付きの文字列で指定することもできます。
	*/
</script>
