<?php
header('Content-type: text/plain; charset=UTF-8');
if(isset($_POST['userid'])&& isset($_POST['password'])){
	$id=$_POST['userid'];
	$pas=$_POST['password'];
	$str="AJAX REQUEST SUCCESS\nuserid:".$id."\npassword:".$pas."\n";
	$result=nl2br($str);
	print $result;
}else{
	print "FAIL TO AJAX REQUEST";
}

scriptの内容

<script>
$(function () {
	//Ajax Button Click
	$('#ajax').on('click', function () {
		$.ajax({
			url: './request.php',
			type: 'POST',	//HTTPのリクエストメソッドを指定。
			data: {
				'userid': $('#userid').val(),
				'password': $('#password').val()
			}
		})
			// Ajaxリクエストが成功した時に発動。
			.done((data) => {
				$('.result').html(data);
				console.log(data);
			})
			//Ajaxリクエストが失敗した時に発動。
			.fail((data) => {
				$('.result').html(data);
				console.log(data);
			})
			//Ajaxリクエストがどうなろうが発動
			.always((data) => {
			});
	});
});
</script>
